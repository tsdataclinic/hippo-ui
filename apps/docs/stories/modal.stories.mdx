export const ModalTrigger = () => {
  const [isOpen, setIsOpen] = useState(false);
  return (
    <>
      <Button onClick={() => setIsOpen(true)}>Hello!</Button>
      <Modal
        title="Test modal"
        isOpen={isOpen}
        onDismiss={() => setIsOpen(false)}
      >
        This is the modal content.
      </Modal>
    </>
  );
};
